% ======================== AI METHODOLOGY ====================================
\section*{\faCogs\ \ AI Analysis Methodology}

\begin{methodbox}
\begin{enumerate}[leftmargin=1.5em, itemsep=4pt, label=\textbf{\arabic*.}]
	\item \textbf{Image Acquisition \& Preprocessing:}
		The chest radiograph is resized to 224$\times$224 pixels, normalized using ImageNet channel statistics
		($\mu$=[0.485, 0.456, 0.406], $\sigma$=[0.229, 0.224, 0.225]), and converted to a 4D tensor for model input.

	\item \textbf{Deep Learning Classification (MedFusionNet):}
		MedFusionNet fuses visual and demographic information: a DenseNet backbone extracts multi-scale
		chest radiograph features, a self-attention module captures long-range spatial dependencies,
		and a Feature Pyramid Network (FPN) aggregates these representations alongside the patient's
		age for joint multi-label classification.
		The model was trained on the NIH ChestX-ray14 dataset (112,120 frontal-view chest radiographs)
		and outputs sigmoid probability scores for {{NUM_CONDITIONS}} thoracic pathologies.

	\item \textbf{Explainability (PyTorch Grad-CAM):}
		Class-specific heatmaps are generated using Gradient-weighted Class Activation Mapping (Grad-CAM)
		applied directly to the MedFusionNet PyTorch checkpoint (\texttt{best\_medfusionnet.pth}).
		Gradients flowing back from the target class logit are used to weight the final convolutional
		feature maps, producing a spatially resolved saliency map that highlights pathologically
		relevant regions in the radiograph.

	\item \textbf{Clinical Reasoning (RAG Pipeline):}
		A Retrieval-Augmented Generation (RAG) pipeline retrieves relevant medical literature from a curated
		knowledge base and synthesizes a structured clinical impression using GPT-4o-mini,
		contextualizing the AI findings within established radiological practice.
\end{enumerate}
\end{methodbox}
